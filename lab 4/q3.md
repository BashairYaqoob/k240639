#include <stdio.h>
int main () {
	char character;
	printf("enter a character: \n");
	scanf(" %c", &character);

	if(character >= 65 && character <= 90) {
		print("capital letter");
	}
	elseif(character >= 32 && character <= 47){
		print("special digit");
	}
	elseif(character >= 97 && character <= 122){
		print("small letter");
	}
	elseif(character >= 48 && character <= 57) {
		print("digit");
	}	

}
